drop table SessionGraphics;
drop table SessionTable;
drop table UserFriend;
drop table UserData;

CREATE TABLE UserData
	(	UserID int auto_increment NOT NULL Primary KEY,
		uEmail varchar(254) NOT NULL UNIQUE,
		uName varchar(128) NOT NULL,
		uPassword varchar(254) NOT NULL,
        verified BOOLEAN NOT NULL DEFAULT FALSE,
        created_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP
	);

CREATE TABLE UserFriend 
	(	FriendPairID int auto_increment NOT NULL PRIMARY KEY,
		UserID int NOT NULL UNIQUE,
		friendID int NOT NULL UNIQUE,
		status ENUM('pending', 'accepted') NOT NULL DEFAULT ('pending'),
        created_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP
	);
ALTER TABLE UserFriend ADD CONSTRAINT UserFriend_UserIDfk FOREIGN KEY (UserID) REFERENCES UserData (UserID) ON DELETE CASCADE;
ALTER TABLE UserFriend ADD CONSTRAINT UserFriend_friendIDfk FOREIGN KEY (friendID) REFERENCES UserData (UserID) ON DELETE CASCADE;

CREATE TABLE SessionTable
	(	SessionID int auto_increment NOT NULL Primary KEY,
		UserID int NOT NULL,
        SessionDate Date NOT NULL,
        SessionLengthTotal int NOT NULL,
        SessionAvgFocus int NOT NULL
    );
ALTER TABLE SessionTable ADD CONSTRAINT SessionTable_UserIDfk FOREIGN KEY (UserID) REFERENCES UserData (UserID) ON DELETE CASCADE;


CREATE TABLE SessionGraphics
	(	SessionGraphicsID int auto_increment NOT NULL Primary KEY,
		SessionID int NOT NULL,
        placeHolder int NOT NULL
	);
ALTER TABLE SessionGraphics ADD CONSTRAINT SessionGraphics_SessionIDfk FOREIGN KEY (SessionID) REFERENCES SessionTable (SessionID) ON DELETE CASCADE;
