drop table SessionChunk;
drop table UserSession;
drop table UserFriend;
drop table UserData;

CREATE TABLE UserData
	(	UserID int auto_increment NOT NULL Primary KEY,
		uEmail varchar(254) NOT NULL UNIQUE,
		uName varchar(128) NOT NULL,
		uPassword varchar(254) NOT NULL,
        verified BOOLEAN NOT NULL DEFAULT FALSE,
        created_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP
	);

CREATE TABLE UserSession
	(	SessionID int auto_increment NOT NULL Primary KEY,
		UserID int NOT NULL,
        sessionStart DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
        sessionEnd DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
        avgFocus double NOT NULL
    );
ALTER TABLE UserSession ADD CONSTRAINT UserSession_UserIDfk FOREIGN KEY (UserID) REFERENCES UserData (UserID) ON DELETE CASCADE;

CREATE TABLE SessionChunk
	(	ChunkId int auto_increment NOT NULL Primary KEY,
		SessionID int NOT NULL,
        UserID int NOT NULL,
        endOfChunk DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
        chunkStatus enum ('VF', 'SF' ,'VU', 'SU')
	);
ALTER TABLE SessionChunk ADD CONSTRAINT SessionChunk_SessionIDfk FOREIGN KEY (SessionID) REFERENCES UserSession (SessionID) ON DELETE CASCADE;
ALTER TABLE SessionChunk ADD CONSTRAINT SessionChunk_UserIDfk FOREIGN KEY (UserID) REFERENCES UserData (UserID) ON DELETE CASCADE;
